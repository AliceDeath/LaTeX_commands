%数列与级数
\newcommand{\series}[1]{\sum\limits_{k=1}^{\infty}{#1}}%只有名字的级数  !!! 格式待修改:不换行要使行间距增大 !!!,1个参数:数列名字
\newcommand{\seq}[1]{$\{ {#1}_{n}\}_{n\ge 1}$}%(只配有名字的)简单的数列; 特别注意!!!:这是用在正文里的,如果在数学环境中使用将会出bug!!!,1个参数:数列名字


%范数和度量函数
\newcommand{\doubVert}[1]{\Vert{#1}\Vert}%习惯性的norm符号:双竖线,1个参数:内容物
\newcommand{\doubvert}[1]{\vert{#1}\vert}%只要输内容物的绝对值(单竖线),1个参数:内容物
\newcommand{\pnorm}[1]{\doubVert{{#1}}_{p}}%pnorm,多见于Euclid空间,1个参数:内容物
\newcommand{\twonorm}[1]{\doubVert{{#1}}_{2}}%一般定义的2norm,多见于Euclid空间,1个参数:内容物
\newcommand{\infnorm}[1]{\doubVert{{#1}}_{\infty}}%一般定义的infity norm,多见于Euclid空间,1个参数:内容物
\newcommand{\twod}[1]{d({#1})_{2}}%一般定义的2度量函数,多见于Euclid空间,1个参数:内容物
\newcommand{\infd}[1]{d({#1})_{\infty}}%一般定义的infity度量函数,多见于Euclid空间,1个参数:内容物
