%%%线性代数板块

%矩阵
\newcommand{\twobm}[4]{\begin{bmatrix}{#1} & {#2} \\ {#3} & {#4} \end{bmatrix}}%一个2*2矩阵,4个参数:按从左到右顺序排
\newcommand{\thrbm}[9]{\begin{bmatrix}{#1} & {#2} & {#3} \\ {#4} & {#5} & {#6} \\{#7} & {#8} & {#9} \end{bmatrix}}%一个3*3矩阵,9个参数:按从左到右顺序排

%特殊矩阵群
\newcommand{\SO}{\mathfrak{so}}%辛方阵
\newcommand{\Hthr}{\mathbb{H}_{3}}%三阶四元数
\newcommand{\Hfour}{\mathbb{H}_{4}}%四阶四元数

%线性空间,线性映射,线性变换
\newcommand{\dualspace}[1]{{#1}^{*}}%对偶空间,1个参数:空间名
\renewcommand{\span}[1]{\langle {\rm {#1}}\rangle}%张出/线性组合出的所有物,1个参数:生成元集(尽量直接打入集合名字,有去除斜体格式的效果)
